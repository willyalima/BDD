Funcionalidade: Entrega Grátis


Oferecida para clientes VIP uma vez que eles comprem um certo 
número de livros

Não é oferecida para clientes comuns e nem para clientes VIPs que comprem qualquer coisa diferente de livros

O número mínimo de livros para a entrega grátis é 5

// api.spec.js

describe('Testes da API', () => {
  it('Deve retornar status code 200, 201 e 400 em diferentes requisições POST', () => {
    const payload = {
      nome: 'Exemplo',
      idade: 30
    };

    cy.request('POST', 'https://api.exemplo.com/recurso', payload)
      .then((response) => {
        expect(response.status).to.eq(200);
      });

    cy.request('POST', 'https://api.exemplo.com/recurso', payload)
      .then((response) => {
        expect(response.status).to.eq(201);
      });

    payload.nome = ''; // Payload inválido
    cy.request({
      method: 'POST',
      url: 'https://api.exemplo.com/recurso',
      body: payload,
      failOnStatusCode: false
    }).then((response) => {
      expect(response.status).to.eq(400);
    });
  });
});
